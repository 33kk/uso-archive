### Description
Hide ads, navigation, sidebar, etc. from livedoor news (http://news.livedoor.com/) photo news page.
This userCSS is for AutoPagerize Greasemonkey Script user.
example: After Install Greasemonkey, AutoPagerize and write following SITEINFO snippets in AutoPagerize userscript, visit here - This page contain Japanese costume-player's photograph. (http://news.livedoor.com/article/image_detail/3780036/?img_id=475442)

<SITEINFO>
var SITEINFO = [
                {
                    url : 'http://news.livedoor.com/article/image_detail/.*',
                    nextLink : '//div[@class="subtitle"]/preceding-sibling::div[@class="photo-link"][1]/ul/li[@class="link-next"]/a[@href]',
                    insertBefore : '//div[@class="subtitle"]',
                    pageElement : 'id("content")',
                    exampleUrl : 'http://news.livedoor.com/article/image_detail/3780036/?img_id=475442',
                },
]
</SITEINFO>
